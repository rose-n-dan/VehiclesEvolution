20 const CarParameters EvolutionaryAlgorithm::generateRandomCar() const {
21  // wheels radiuses generation
22  std::normal_distribution<double> wheel_radius_distribution(0.5, 0.15);
23  double first_wheel_radius = std::fabs(wheel_radius_distribution(generator_));
24  double second_wheel_radius = std::fabs(wheel_radius_distribution(generator_));
25 
26  std::uniform_real_distribution<double> car_body_distribution_theta(0.0, 2 * M_PI / CarParameters::NUMBER_OF_CAR_BODY_POINTS_);
27  std::uniform_real_distribution<double> car_body_distribution_radius(0.3, 2);
28  std::vector<b2Vec2> car_body_points;
29  for (int i = 0; i < CarParameters::NUMBER_OF_CAR_BODY_POINTS_; i++) {
30  double offset = i * 2 * M_PI / CarParameters::NUMBER_OF_CAR_BODY_POINTS_;
31  double theta = offset + car_body_distribution_theta(generator_);
32  double radius = car_body_distribution_radius(generator_);
33  double x = radius * cos(theta);
34  double y = radius * sin(theta);
35  car_body_points.emplace_back(x, y);
36  }
37 
38  std::uniform_int_distribution<int> joint_index_distribution(0, CarParameters::NUMBER_OF_CAR_BODY_POINTS_ - 1);
39  int first_joint_index = joint_index_distribution(generator_);
40  int second_joint_index = joint_index_distribution(generator_);
41  while (first_joint_index == second_joint_index) {
42  second_joint_index = joint_index_distribution(generator_);
43  }
44  b2Vec2 first = car_body_points.at(first_joint_index);
45  b2Vec2 second = car_body_points.at(second_joint_index);
46  b2Vec2 front_joint;
47  b2Vec2 rear_joint;
48  if (first.x < second.x) {
49  front_joint = second;
50  rear_joint = first;
51  }
52  else {
53  front_joint = first;
54  rear_joint = second;
55  }
56 
57  return CarParameters(first_wheel_radius, second_wheel_radius, car_body_points, front_joint, rear_joint);
58 }

70 const std::vector<CarParameters> EvolutionaryAlgorithm::makeNewGeneration(const std::vector<double> &distances) {
71  // add distances to last_generation_parameters vector
72  {
73  auto parIt = last_generation_parameters_.begin();
74  auto distIt = distances.begin();
75  for (; parIt != last_generation_parameters_.end() && distIt != distances.end(); parIt++, distIt++) {
76  parIt->distance_ = *distIt;
77  }
78  }
79 
80  // sort last_generation_parameters by distance
81  std::sort(last_generation_parameters_.begin(), last_generation_parameters_.end(),
82  [](const CarParameters & a, const CarParameters & b) -> bool
83  {
84  return a.distance_ < b.distance_;
85  });
86 
87  // newly generated population
88  std::vector<CarParameters> newly_generated_population;
89 
90  // elitist selection - two best guys have guaranteed places in new population
91  newly_generated_population.push_back(last_generation_parameters_.rbegin()[0]);
92 // newly_generated_population.push_back(last_generation_parameters_.rbegin()[1]);
93 
94  // rank based selection
95  while (newly_generated_population.size() < distances.size()) {
96  // crossover - 100%
97  CarParameters new_car = doCrossover(last_generation_parameters_.at(selectOneInN(CAR_NUMBER_IN_POPULATION_)),
98  last_generation_parameters_.at(selectOneInN(CAR_NUMBER_IN_POPULATION_)));
99  // mutation
100  maybeMutate(new_car);
101  newly_generated_population.push_back(new_car);
102  }
103 
104  // store newly generated population as last generation
105  last_generation_parameters_.clear();
106  last_generation_parameters_ = newly_generated_population;
107 
108  return newly_generated_population;
109 }

166 CarParameters EvolutionaryAlgorithm::doCrossover(const CarParameters & mother, const CarParameters & father) const {
167  //create new parameters
168  CarParameters new_car_parameters;
169  //crossover on wheel radius
170  new_car_parameters.front_wheel_radius_ = doCrossover(mother.front_wheel_radius_, father.front_wheel_radius_);
171  new_car_parameters.rear_wheel_radius_ = doCrossover(mother.rear_wheel_radius_, father.rear_wheel_radius_);
172  //crossover on carbody points
173  for (int i = 0; i < CarParameters::NUMBER_OF_CAR_BODY_POINTS_; i++) {
174  new_car_parameters.car_body_[i].x = doCrossover(mother.car_body_[i].x, father.car_body_[i].x);
175  new_car_parameters.car_body_[i].y = doCrossover(mother.car_body_[i].y, father.car_body_[i].y);
176  }
177  //pick right joint position
178  //picking from bottom area of carbody
179  std::uniform_int_distribution<int> joint_index_distribution(3, 5);
180  int first_joint_index = joint_index_distribution(generator_);
181  int second_joint_index = joint_index_distribution(generator_);
182  while (first_joint_index == second_joint_index) {
183  second_joint_index = joint_index_distribution(generator_);
184  }
185  if(first_joint_index > second_joint_index) {
186  new_car_parameters.front_joint_ = new_car_parameters.car_body_.at(second_joint_index);
187  new_car_parameters.rear_joint_ = new_car_parameters.car_body_.at(first_joint_index);
188  }
189  else {
190  new_car_parameters.front_joint_ = new_car_parameters.car_body_.at(first_joint_index);
191  new_car_parameters.rear_joint_ = new_car_parameters.car_body_.at(second_joint_index);
192  }
193 
194  return new_car_parameters;
195 }
